CREATE DATABASE IF NOT EXISTS StudyBudyDB;
USE StudyBudyDB;

DROP TABLE Download;
DROP TABLE Comment;
DROP TABLE Post;
DROP TABLE Summary;
DROP TABLE Student;

#DROP TABLE Student;
CREATE TABLE Student (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Email VARCHAR(255) NOT NULL,
    Password VARCHAR(255) NOT NULL,
    Nickname VARCHAR(255),
    Phone_number VARCHAR(20),
    Credit INT DEFAULT 0 CHECK (Credit >= 0)
);
#DROP TABLE Summary;
CREATE TABLE Summary (
    Summary_ID INT AUTO_INCREMENT PRIMARY KEY,
    Name_Summary VARCHAR(255),
    Course_Number VARCHAR(255),
    Course_Name VARCHAR(255),
    teacher VARCHAR(255),
    Year VARCHAR(255),
    Semester VARCHAR(255),
    numDownloads INT DEFAULT 0 CHECK (numDownloads >= 0), ## needs to be deleted
    uploadDate DATE,
    summaryUrl VARCHAR(255),
    uploader_id INT,
    FOREIGN KEY (uploader_id) REFERENCES Student(ID)
);
#DROP TABLE Post;
CREATE TABLE Post (
    Post_ID INT AUTO_INCREMENT PRIMARY KEY,
    DateTime DATETIME,
    Text TEXT,
	Poster_ID INT,
    FOREIGN KEY (Poster_ID) REFERENCES Student(ID)
);
#DROP TABLE Comment;
CREATE TABLE Comment (
    Comment_ID INT AUTO_INCREMENT PRIMARY KEY,
    DateTime DATETIME,
    Text TEXT,
    Post_ID INT,
	Student_ID INT,
    FOREIGN KEY (Post_ID) REFERENCES Post(Post_ID),
    FOREIGN KEY (Student_ID) REFERENCES Student(ID)
);

#DROP TABLE Download;
CREATE TABLE Download (
    Student_ID INT,
    Summary_ID INT,
    Date_of_download DATE,
    PRIMARY KEY (Student_ID, Summary_ID),
    FOREIGN KEY (Student_ID) REFERENCES Student(ID),
    FOREIGN KEY (Summary_ID) REFERENCES Summary(Summary_ID)
);
